package datamodel;


/**
 * Immutable entity class representing an <i>Article</i>, which is a product
 * referred to in orders.
 * <br>
 * An <i>immutable</i> class does not allow changes to attributes.
 * {@link DataFactory} is the only class that creates {@link Article}
 * objects from validated arguments.
 * 
 * @version <code style=color:green>{@value application.package_info#Version}</code>
 * @author <code style=color:blue>{@value application.package_info#Author}</code>
 */
public final class Article {

    /**
     * Unique Article identifier. Must not be {@code null}.
     */
    private final String id;

    /**
     * Article description. Must not be {@code null} and not empty {@code ""}.
     */
    private final String description;

    /**
     * None-public constructor used by {@link DataFactory} preventing object
     * creation outside this package.
     * @param id article identifier supplied by {@link DataFactory}
     * @param description article description
     * @throws IllegalArgumentException if arguments are {@code null} or
     * description is empty {@code ""}
     */
    protected Article(String id, String description) {
        if(id==null)
            throw new IllegalArgumentException("id null");
        if(description==null || description.length()==0)
            throw new IllegalArgumentException("description null or empty");
        //
        this.id = id;
        this.description = description;
    }

    /**
     * Id attribute getter.
     * @return article id
     */
    public String getId() {
        return id;
    }

    /**
     * Description attribute getter.
     * @return article description
     */
    public String getDescription() {
        return description;
    }
}
