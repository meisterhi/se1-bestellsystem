package datamodel;

import java.time.LocalDateTime;
import java.util.ArrayList;
import java.util.List;
import java.util.stream.StreamSupport;


/**
 * Immutable entity class representing an <i>Order</i>, which is a
 * collection of {@link Article} items purchased by a {@link Customer}.
 * <br>
 * An <i>immutable</i> class does not allow changes to attributes.
 * {@link DataFactory} is the only class that creates {@link Order}
 * objects from validated arguments.
 * 
 * @version <code style=color:green>{@value application.package_info#Version}</code>
 * @author <code style=color:blue>{@value application.package_info#Author}</code>
 */
public final class Order {

    /**
     * Record of an order line item comprised of a number of units order
     * of one {@link Article}.
     * @param article ordered article
     * @param unitsOrdered ordered number of units
     */
    public final record OrderItem (
        Article article,
        long unitsOrdered
    ) {
        /**
         * @hidden
         */
        @Override public int hashCode() { return 0; }
    }

    /**
     * Unique Order id attribute. Must be {@code > 0}).
     */
    private final long id;

    /**
     * Reference to {@link Customer} owning the order. Must not be {@code null}.
     */
    private final Customer customer;

    /**
     * Reference to price {@link OrderItem} objects (country, currency, tax rates).
     * Must not be {@code null}.
     */
    private final Pricing pricing;

    /**
     * Day and time (in UTC) when order was created. Must not be {@code null} and
     * between dates: {@code "Jan 01, 2020 00:00"} and {@code "Dec 31, 2099 23:59"}
     */
    private final LocalDateTime created;

    /**
     * List of {@link OrderItem} objects that constitute this order.
     */
    private final List<OrderItem> orderItems = new ArrayList<>();


    /**
     * None-public constructor used by {@link DataFactory} avoiding object
     * creation outside the package.
     * @param id order identifier supplied by {@link DataFactory}
     * @param customer reference to owning customer (must not be {@code null})
     * @param pricing reference to pricing (country, currency, tax rates) that applies to order (must not be {@code null})
     * @param created day and time (in UTC) when order was created (must not be {@code null})
     * @throws IllegalArgumentException if {@code id} is negative,
     * arguments are {@code null} or creation date is outside bounds
     * between {@code "Jan 01, 2020 00:00"} and {@code "Dec 31, 2099 23:59"}
     */
    protected Order(long id, Customer customer, Pricing pricing, LocalDateTime created) {
        if(id < 0L)
            throw new IllegalArgumentException("id negative");
        if(pricing==null)
            throw new IllegalArgumentException("pricing null");
        if(created==null)
            throw new IllegalArgumentException("created null");
        //
        var validCreationDate = DataFactory.getInstance().validateOrderCreationDate(created);
        if(validCreationDate.isEmpty())
            throw new IllegalArgumentException("creation date outside valid range");
        //
        this.id = id;
        this.customer = customer;
        this.pricing = pricing;
        this.created = validCreationDate.get();
    }

    /**
     * Id attribute getter.
     * @return order id (always {@code > 0})
     */
    public long getId() {
        return id;
    }

    /**
     * Owning customer getter.
     * @return customer owning this order (never {@code null}).
     */
    public Customer getCustomer() {
        return customer;
    }

    /**
     * Return pricing information that applies to this order.
     * @return reference to pricing information (never {@code null})
     */
    public Pricing getPricing() {
        return pricing;
    }

    /**
     * Return day and time (in UTC) when order was created.
     * @return day and time (in UTC) when order was created (never {@code null} and
     * between dates: {@code "Jan 01, 2020 00:00"} and {@code "Dec 31, 2099 23:59"})
     */
    public LocalDateTime getCreated() {
        return created;
    }

    /**
     * Return number of items included in this order.
     * @return number of items included in this order
     */
    public long itemsCount() {
        return orderItems.size();
    }

    /**
     * Return items included in this order.
     * @return items included in this order (as immutable {@link Iterable<String>})
     */
    public Iterable<OrderItem> getOrderItems() {
        return orderItems;
    }

    /**
     * Create and add new item to this order from valid arguments.
     * @param article article to order
     * @param unitsOrdered number of articles to order
     * @return chainable self-reference
     * @throws IllegalArgumentException when article argument is null
     */
    public Order addItem(Article article, long unitsOrdered) {
        if(article==null)
            throw new IllegalArgumentException("argument article null");
        //
        var item = new OrderItem(article, unitsOrdered);
        orderItems.add(item);
        return this;
    }

    /**
     * Delete the i-th line item from order with {@code i >= 0} and {@code i < itemsCount()}.
     * Method has no effect when {@code i} is outside valid bounds.
     * @param i index of line item to delete from order
     */
    public void deleteItem(int i) {
        if(i >= 0 && i < itemsCount()) {
            orderItems.remove(i);
        }
    }

    /**
     * Bulk delete of line items provided by a collection of indices.
     * Only line items with {@code i >= 0} and {@code i < itemsCount()}
     * are deleted. Method has no effect for {@code i} outside valid bounds.
     * @param indices collection of indices of line items to delete from order
     */
    public void deleteItems(Iterable<Integer> indices) {
        var size = orderItems.size();
        var items = StreamSupport.stream(indices.spliterator(), false)
            .filter(i -> i >= 0 && i < size)
            .map(i -> orderItems.get(i))
            .toList();
        //
        // remove items as one transaction from list
        orderItems.removeAll(items);
    }
}
